const page = dv.page(dv.current().file.path) ?? {};
const L = dv.luxon;

const exclude = new Set(["file", "position", "type"]);
const isEmpty = v => v === null || v === undefined ||
  (typeof v === "string" && v.trim() === "") ||
  (Array.isArray(v) && v.length === 0) ||
  (typeof v === "object" && !Array.isArray(v) && Object.keys(v).length === 0);

function fmtDate(val) {
  if (val == null) return "";
  try {
    if (typeof val === "object" && typeof val.toISO === "function") {
      const d = L.DateTime.fromISO(val.toISO());
      if (d.isValid) return d.toFormat("yyyy-LL-dd");
    }
    const s = String(val);
    const d = L.DateTime.fromISO(s);
    if (d.isValid) return d.toFormat("yyyy-LL-dd");
    const m = s.match(/^(\d{4}-\d{2}-\d{2})/);
    return m ? m[1] : s;
  } catch (e) { return String(val); }
}
function isDateLike(val) {
  if (val && typeof val === "object" && typeof val.toISO === "function") return true;
  const d = L.DateTime.fromISO(String(val));
  return d.isValid;
}

function renderValue(val, key) {
  if (key === "tags" || key === "标签") {
    const arr = Array.isArray(val) ? val : String(val).split(/[ ,;]+/).filter(Boolean);
    const frag = document.createDocumentFragment();
    arr.forEach((t, i) => {
      const tag = String(t).replace(/^#+/, "");
      if (i > 0) frag.append(" ");
      const a = document.createElement("a");
      a.className = "tag";
      a.href = `#${tag}`;
      a.textContent = `#${tag}`;
      frag.append(a);
    });
    return frag;
  }
  if (isDateLike(val)) return fmtDate(val);
  if (val && typeof val === 'object' && 'path' in val) return dv.fileLink(val.path);
  if (Array.isArray(val)) return val.map(v => renderValue(v, key)).join("、");
  return String(val);
}

const wrap = dv.el("div", "", { cls: "dvjs-profile-card" });
const table = document.createElement("table");
// 为档案卡片表格添加专用类，便于 CSS 统一控制
table.classList.add("dvjs-profile-table");
Object.assign(table.style, { border: "none", borderCollapse: "collapse", width: "auto" });
wrap.appendChild(table);
// 显式创建 tbody，避免某些环境下 tBodies 为空导致 insertBefore 报错
const tbody = document.createElement("tbody");
table.appendChild(tbody);

function addRow(label, valueNode) {
  const tr = document.createElement("tr");
  const tdL = document.createElement("td");
  const tdR = document.createElement("td");
  tdL.textContent = String(label);
  tdL.style.whiteSpace = "nowrap";
  tdL.style.padding = "2px 8px 2px 0";
  tdR.style.padding = "2px 0";
  tdR.style.minWidth = "10em";
  if (valueNode instanceof Node) tdR.appendChild(valueNode); else tdR.textContent = String(valueNode);
  tr.append(tdL, tdR);
  tbody.appendChild(tr);
}

const entries = Object.entries(page).filter(([k, v]) => !exclude.has(k) && !isEmpty(v));
let birthdayIndex = -1;
entries.forEach(([k, v]) => {
  addRow(k, renderValue(v, k));
  if (k === "生日" || k === "birthday") birthdayIndex = tbody.rows.length - 1;
});

const bRaw = page["生日"] ?? page["birthday"];
if (!isEmpty(bRaw)) {
  let b = null;
  if (bRaw && typeof bRaw === 'object' && typeof bRaw.toISO === 'function') b = L.DateTime.fromISO(bRaw.toISO());
  else b = L.DateTime.fromISO(String(bRaw));
  if (b && b.isValid) {
    const age = Math.floor(L.DateTime.local().diff(b, "years").years);
    const tr = document.createElement("tr");
    const tdL = document.createElement("td");
    const tdR = document.createElement("td");
    tdL.textContent = "年龄";
    tdL.style.whiteSpace = "nowrap";
    tdL.style.padding = "2px 8px 2px 0";
    tdR.textContent = `${age} 岁`;
    tdR.style.padding = "2px 0";
    tr.append(tdL, tdR);
    // 插在生日行之后；若未找到生日行则追加到末尾
    const nextRow = tbody.rows[birthdayIndex + 1];
    if (birthdayIndex >= 0 && nextRow) tbody.insertBefore(tr, nextRow);
    else tbody.appendChild(tr);
  }
}
