## 与会者

```dataviewjs
// 在 Dynbedded 环境下统一使用 dv.current() 作为宿主页面
const host = dv.current();
// 通过 dv.page 获取完整的页面字段（含 frontmatter 转换后的值）
const page = dv.page(host.file.path) ?? {};
// attendees 可能是 Link 对象数组或字符串数组，统一转为可解析的字符串
const raw = page.attendees ?? [];
const attendees = Array.isArray(raw) ? raw : [];

const container = dv.el('div', null, { cls: 'dvjs-event-attendees' });
const list = document.createElement('ul');
list.className = 'attendees-list';
container.appendChild(list);

function makeInternalLink(item) {
  // 支持三种形式：Link 对象、"[[路径|别名]]" 字符串、普通字符串（文件名/路径）
  let targetPath = null;
  let alias = null;
  if (item && typeof item === 'object' && 'path' in item) {
    targetPath = String(item.path);
    // Dataview Link 无别名信息，这里用文件名作为显示文本
    const p = dv.page(targetPath);
    alias = p ? p.file.name : null;
  } else {
    const s = String(item ?? '').trim();
    const m = s.match(/^\[\[([^|\]]+)(?:\|([^\]]+))?\]\]$/);
    if (m) {
      targetPath = m[1].trim();
      alias = m[2] ? m[2].trim() : null;
    } else {
      targetPath = s;
    }
  }
  const pageObj = targetPath ? dv.page(targetPath) : null;
  const text = alias || (pageObj ? pageObj.file.name : (targetPath || ''));
  if (!pageObj) {
    // 找不到页面时，仅返回文本节点，避免产生无效链接
    const span = document.createElement('span');
    span.textContent = text;
    return span;
  }
  const a = document.createElement('a');
  a.classList.add('internal-link');
  a.href = pageObj.file.path;
  a.setAttribute('data-href', pageObj.file.path);
  a.textContent = text || pageObj.file.name;
  return a;
}

for (const it of attendees) {
  const li = document.createElement('li');
  li.appendChild(makeInternalLink(it));
  list.appendChild(li);
}

if (attendees.length === 0) {
  const li = document.createElement('li');
  li.textContent = '暂无与会者';
  list.appendChild(li);
}

dv.container.appendChild(container);
```
