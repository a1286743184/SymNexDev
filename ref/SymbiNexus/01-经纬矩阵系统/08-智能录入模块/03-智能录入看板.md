// 1. 引入 SmartInputKit 核心库 v2.0
const kitPath = '08-科技树系统/01-OB_LifeOS_Build/00-通用模块库/SmartInputKit.js';
const kitFile = app.vault.getAbstractFileByPath(kitPath);

if (!kitFile) {
    dv.paragraph(`❌ 致命错误：核心库未找到，请检查路径：${kitPath}`);
} else {
    // 2. 加载并执行库文件
    const code = await app.vault.read(kitFile);
    const scriptFunc = new Function("window", code);
    scriptFunc(window);
    
    // 等待挂载
    await new Promise(r => setTimeout(r, 50));

    if (window.SmartInputKit) {
        // 3. 执行主程序 (Phase 2 Update)
        // main 方法会自动处理 概览渲染 和 图表渲染
        window.SmartInputKit.render.main(dv);
    } else {
        dv.paragraph("❌ 错误：SmartInputKit 加载失败");
    }
}
