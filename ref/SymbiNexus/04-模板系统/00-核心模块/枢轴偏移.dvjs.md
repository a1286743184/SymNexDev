// DataviewJS 版：仅导出 3 点枢轴逻辑到 window.ObLife.pivot3am
// 不进行任何渲染或 frontmatter 写入，避免副作用
(() => {
  const L = dv.luxon;
  const DEFAULT_PIVOT_HOUR = 3;

  // 始终将输入统一转换为“本地时区”的 Luxon DateTime
  const toLuxon = (x) => {
    try {
      if (!x) return L.DateTime.local();
      // Dataview/Tasks 的日期对象（多数为 Luxon）
      if (x && typeof x === 'object' && typeof x.toISODate === 'function') {
        const d = L.DateTime.fromISO(x.toISO?.() ?? x.toISODate(), { setZone: true });
        return d?.isValid ? d.toLocal() : L.DateTime.local();
      }
      // 原生 Date
      if (x instanceof Date) return L.DateTime.fromJSDate(x).toLocal();
      // 字符串（可能带 Z/时区信息）
      if (typeof x === 'string') {
        const byISO = L.DateTime.fromISO(x, { setZone: true });
        if (byISO?.isValid) return byISO.toLocal();
        return L.DateTime.fromJSDate(new Date(x)).toLocal();
      }
    } catch (e) {}
    return L.DateTime.local();
  };

  // 检测是否“仅日期”（无明确时间部分）：
  // 1) 纯日期字符串：YYYY-MM-DD
  // 2) Luxon/Date 对象在本地时区为 00:00:00.000（多数来自 Dataview 的 date-only）
  const isDateOnly = (raw, dt) => {
    try {
      if (typeof raw === 'string' && /^\d{4}-\d{2}-\d{2}$/.test(raw)) return true;
      const d = dt ?? toLuxon(raw);
      return d.hour === 0 && d.minute === 0 && d.second === 0 && d.millisecond === 0;
    } catch (e) { return false; }
  };

  // 关键修复：
  // - 统一按“本地时区”计算枢轴（避免 UTC 字符串在本地被判为凌晨从而错误前移一天）
  // - 对“仅日期”值不应用枢轴偏移（视为该自然日），避免 00:00 被当成枢轴前而前移
  const isoDateWithPivot = (raw, pivotHour = DEFAULT_PIVOT_HOUR) => {
    const dLocal = toLuxon(raw); // 本地时区时间
    if (isDateOnly(raw, dLocal)) {
      return dLocal.startOf('day').toISODate();
    }
    const logical = dLocal.hour < pivotHour ? dLocal.minus({ days: 1 }) : dLocal;
    return logical.startOf('day').toISODate();
  };

  const logicalTodayISO = (pivotHour = DEFAULT_PIVOT_HOUR) => isoDateWithPivot(L.DateTime.local(), pivotHour);

  const startOfLogicalDay = (base = L.DateTime.local(), pivotHour = DEFAULT_PIVOT_HOUR) => {
    const d = base.hour < pivotHour ? base.minus({ days: 1 }) : base;
    return d.startOf('day').plus({ hours: pivotHour });
  };

  const endOfLogicalDay = (base = L.DateTime.local(), pivotHour = DEFAULT_PIVOT_HOUR) => startOfLogicalDay(base, pivotHour).plus({ days: 1 });

  const belongsToLogicalDate = (dt, isoDate, pivotHour = DEFAULT_PIVOT_HOUR) => isoDateWithPivot(dt, pivotHour) === isoDate;

  const isCancelledLine = (text) => /^❌\s*\d{4}-\d{2}-\d{2}/.test(text ?? '');

  // 新增：用于与 Dataview 任务的 due/completion 对接的桥接函数
  const completionISO = (completion, pivotHour = DEFAULT_PIVOT_HOUR) => isoDateWithPivot(completion, pivotHour);
  const completionDT = (completion, pivotHour = DEFAULT_PIVOT_HOUR) => {
    const iso = completionISO(completion, pivotHour);
    try {
      return iso ? L.DateTime.fromISO(iso).startOf('day') : null;
    } catch (e) { return null; }
  };
  const _startOfDay = (d) => (d && typeof d.startOf === 'function') ? d.startOf('day') : d;
  // 修复逾期判定：必须比较“枢轴日”而非具体时间
  const isLateByPivot = (due, completion, pivotHour = DEFAULT_PIVOT_HOUR) => {
    try {
      const dueISO = isoDateWithPivot(due, pivotHour);
      const compISO = isoDateWithPivot(completion, pivotHour);
      if (!dueISO || !compISO) return false;
      // ISO 日期字符串可直接按字典序比较先后
      return compISO > dueISO;
    } catch (e) { return false; }
  };

  // 检查任务是否在逻辑今天完成（处理枢轴偏移边界情况）
  const isCompletedToday = (task, pivotHour = DEFAULT_PIVOT_HOUR) => {
    const compISO = completionISO(task?.completion, pivotHour);
    const todayISO = logicalTodayISO(pivotHour);
    return compISO === todayISO;
  };

  // 检查任务是否在本周完成（处理枢轴偏移边界情况）
  const isCompletedThisWeek = (task, weekStartISO, weekEndISO, pivotHour = DEFAULT_PIVOT_HOUR) => {
    const compISO = completionISO(task?.completion, pivotHour);
    if (!compISO) return false;
    
    // 标准范围检查
    return compISO >= weekStartISO && compISO <= weekEndISO;
  };

  // 简化页面调用：返回绑定了 pivotHour 的便捷函数集合
  const bind = (pivotHour = DEFAULT_PIVOT_HOUR) => ({
    todayISO: logicalTodayISO(pivotHour),
    completionISO: (x) => completionISO(x, pivotHour),
    completionDT: (x) => completionDT(x, pivotHour),
    isLateByPivot: (d, c) => isLateByPivot(d, c, pivotHour),
    isCompletedToday: (t) => isCompletedToday(t, pivotHour),
    isCompletedThisWeek: (task, weekStartISO, weekEndISO) => isCompletedThisWeek(task, weekStartISO, weekEndISO, pivotHour),
  });

  // 可选：向全局注入便捷函数，避免每页重复解构（不修改 Luxon/Dataview 原型，保证安全）
  const injectGlobals = (pivotHour = DEFAULT_PIVOT_HOUR) => {
    try {
      window.pivot3amTodayISO = logicalTodayISO(pivotHour);
      window.pivot3amCompletionISO = (x) => completionISO(x, pivotHour);
      window.pivot3amCompletionDT = (x) => completionDT(x, pivotHour);
      window.pivot3amIsLate = (d, c) => isLateByPivot(d, c, pivotHour);
    } catch (e) {}
  };

  window.ObLife = window.ObLife || {};
  window.ObLife.pivot3am = {
    DEFAULT_PIVOT_HOUR,
    isoDateWithPivot,
    logicalTodayISO,
    startOfLogicalDay,
    endOfLogicalDay,
    belongsToLogicalDate,
    isCancelledLine,
    completionISO,
    completionDT,
    isLateByPivot,
    isCompletedToday,
    isCompletedThisWeek,
    bind,
    injectGlobals,
  };
})();
