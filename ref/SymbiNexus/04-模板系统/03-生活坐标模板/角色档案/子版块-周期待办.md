const host = dv.current();
const name = host.file.name;
const L = dv.luxon;

// 目标来源目录：周委托与规律性事项（按需取消外部汇总）
const pages = [];
const localTasks = host.file.tasks || [];
const currentPath = host.file.path;

// 收集并过滤任务：仅当前页面的未完成任务
let tasks = (host.file.tasks || []).filter(t => !t.completed);

// 按到期日从近到远排序（无到期日置后）
const toMillis = (d) => {
  if (!d) return null;
  try {
    return typeof d.toMillis === 'function' ? d.toMillis() : L.DateTime.fromISO(String(d)).toMillis();
  } catch (e) {
    return null;
  }
};

// 基于“距离今天的距离”进行排序：最近的未来日期优先，其次是已过期的（但仍未完成）任务，最后是无日期任务
const nowMs = L.DateTime.now().toMillis();
const rank = (m) => {
  if (m == null) return Number.POSITIVE_INFINITY; // 无日期排到最后
  const delta = m - nowMs;
  return delta >= 0 ? delta : Math.abs(delta) + 1e12; // 未来优先；过去任务靠后但仍按距离排序
};

tasks.sort((a, b) => {
  const ar = rank(toMillis(a.due));
  const br = rank(toMillis(b.due));
  return ar - br;
});

// 新增：同名任务（忽略日期/循环/标签）只保留最近的一个
const normalizeText = (t) => {
  let s = String(t?.text ?? '');
  // 移除任务插件的常见字段与日期展示
  s = s.replace(/[📅⏳🛫🗓️⏰⏱️]/g, ' '); // 常见图标占位
  s = s.replace(/\b\d{4}-\d{2}-\d{2}\b/g, ' '); // ISO 日期
  s = s.replace(/\bevery\b[^#\n]*/gi, ' '); // 循环描述（every ...）
  s = s.replace(/#[^\s#]+/g, ' '); // 标签
  s = s.replace(/\s+/g, ' ').trim().toLowerCase();
  return s;
};

const seen = new Set();
const deduped = [];
for (const t of tasks) {
  const key = normalizeText(t);
  if (!key) { // 防御性处理
    deduped.push(t);
    continue;
  }
  if (seen.has(key)) continue; // 已有同名任务，跳过
  seen.add(key);
  deduped.push(t); // 排序后最靠前的即“最近”的一个
}

tasks = deduped;

// 使用 Dataview 的任务列表渲染
if (tasks.length === 0) {
  dv.el("div", "暂无匹配的待办任务", { cls: "dvjs-empty" });
} else {
  dv.taskList(tasks, false);
}
