const host = dv.current();
const L = dv.luxon;

// 数据源目录（相对 Vault 路径）
const DIR_EVENTS = "05-生活坐标系统/02-事件记录";
const DIR_FOOTPRINTS = "02-复盘系统/02-日复盘";

// 辅助函数
const toMillis = (d) => {
  if (!d) return null;
  try {
    return typeof d.toMillis === 'function' ? d.toMillis() : L.DateTime.fromISO(String(d)).toMillis();
  } catch (e) { return null; }
};
const fmtDate = (val) => {
  if (!val) return "";
  try {
    if (typeof val === "object" && typeof val.toISO === "function") {
      const d = L.DateTime.fromISO(val.toISO());
      return d.isValid ? d.toFormat("yyyy-LL-dd") : String(val);
    }
    const d = L.DateTime.fromISO(String(val));
    return d.isValid ? d.toFormat("yyyy-LL-dd") : String(val);
  } catch (e) { return String(val); }
};
// 去除括号内容与所有数字（含半角/全角括号）并清理多余空格
const cleanTitle = (name) => {
  let s = String(name ?? "");
  s = s.replace(/[\(\（][^\)\）]*[\)\）]/g, ""); // 移除括号内内容
  s = s.replace(/[0-9]/g, ""); // 移除数字
  s = s.replace(/\s{2,}/g, " ").trim();
  return s;
};
const hasLink = (arr, link) => {
  if (!arr || !link) return false;
  const target = link.path ?? String(link);
  return Array.isArray(arr) && arr.some(x => {
    const xp = (x && x.path) ? x.path : String(x);
    return xp === target;
  });
};

// 收集页面
const events = dv.pages(`"${DIR_EVENTS}"`);
const footprints = dv.pages(`"${DIR_FOOTPRINTS}"`);
const all = events.concat(footprints).array();

// 仅纳入与当前角色有关的记录（包含该角色的双链或 attendees 字段包含该角色）
const linked = all.filter(p => hasLink(p.file.outlinks, host.file.link) || hasLink(p.attendees, host.file.link));

// 映射为渲染行并按日期倒序（最新在上）
const rows = linked
  .map(p => ({
    title: p.file.name,
    summary: p.summary ?? "",
    date: p.date ?? null,
    page: p
  }))
  .sort((a, b) => (toMillis(b.date) ?? 0) - (toMillis(a.date) ?? 0));

// 渲染 HTML 表格（无边框、首行紫色加粗）
const root = dv.el("div", "", { cls: "dvjs-profile-timeline" });
if (rows.length === 0) {
  dv.el("div", "暂无时间轴记录", { el: root, cls: "dvjs-empty" });
} else {
  const table = dv.el("table", "", { el: root, cls: "dvjs-profile-timeline-table" });
  const thead = table.createTHead();
  const hr = thead.insertRow();
  ["事件", "简介", "日期"].forEach(h => {
    const th = document.createElement("th");
    th.textContent = h;
    hr.appendChild(th);
  });
  const tbody = document.createElement("tbody");
  table.appendChild(tbody);

  rows.forEach(r => {
    const tr = document.createElement("tr");
    const tdTitle = document.createElement("td");
    // 使用 Obsidian 内部链接元素，确保悬浮预览与跳转，显示为 [[文件名]] 的简洁样式
    const a = document.createElement("a");
    a.className = "internal-link";
    a.dataset.href = r.page.file.path;
    a.href = r.page.file.path;
    a.textContent = r.title;
    a.target = "_blank";
    a.rel = "noopener";
    tdTitle.appendChild(a);

    const tdSummary = document.createElement("td");
    tdSummary.textContent = r.summary ?? "";

    const tdDate = document.createElement("td");
    tdDate.textContent = fmtDate(r.date);

    tr.append(tdTitle, tdSummary, tdDate);
    tbody.appendChild(tr);
  });
}
