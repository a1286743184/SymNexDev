// 引入 Templater 版 3 点枢轴函数（仅函数，无说明/示例，避免副作用）
// 严格单入口管理：必须从模块获取 pivotDateISO，不提供本地兜底
await tp.file.include("[[枢轴偏移.tpl]]");

const date = tp.date.now("YY-MM-DD");
// 更稳健的重命名：若同名文件已存在则自动加序号避免报错
const baseName = `${date}事件记录`;
const file = app.workspace.getActiveFile();
const folder = file?.parent?.path ?? tp.file.folder(false);
const exists = (name) => !!app.vault.getAbstractFileByPath(`${folder}/${name}.md`);
let targetName = baseName;
if (exists(targetName)) {
  let i = 2;
  while (exists(`${baseName}-${i}`)) i++;
  targetName = `${baseName}-${i}`;
}
try {
  await tp.file.rename(targetName);
} catch (e) {
  console.warn("重命名失败，将跳过重命名继续执行模板", e);
}

// 一次性设置 Frontmatter 的日期属性（使用模块提供的 pivotDateISO(3)）
// 注意：避免在 DataviewJS 中写 frontmatter，防止渲染-写入循环。
try {
  const f = app.workspace.getActiveFile();
  if (f) {
    const logicalISO = pivotDateISO(3); // 严格单入口：直接调用模块函数
    app.fileManager.processFrontMatter(f, fm => {
      if (!fm.date || fm.date !== logicalISO) fm.date = logicalISO; // 缺失或不一致时写入
    });
  }
} catch (e) {
  console.warn("设置 frontmatter 逻辑日期失败", e);
}

// 触发 Obsidian 命令：属性视图 -> 显示当前笔记的属性
// 兼容中英文界面与不同版本；并提供提示信息
try {
  const sleep = (ms) => new Promise(res => setTimeout(res, ms));
  // 稍等一下，避免模板刚创建/重命名后命令未就绪
  await sleep(300);
  const cmds = Object.values(app?.commands?.commands ?? {});
  // 首选：属性列表（你截图中显示的命令）
  const listNames = [
    "属性列表: 显示当前笔记的属性列表",
    "属性列表：显示当前笔记的属性列表",
    "Properties: Show current note's properties",
    "Properties: Show current note's properties list",
    "Properties: Show current file's properties list",
  ];
  const listMatch = cmds.find(c => listNames.includes(c.name))
    || cmds.find(c => (c.name || "").includes("属性列表"))
    || cmds.find(c => ((c.id || "").includes("properties") && (c.id || "").includes("list")));

  // 备选：属性视图
  const viewIds = [
    "properties-view:show-file-properties",
    "properties-view:open-file-properties",
  ];
  const viewNames = [
    "Properties view: Show file properties",
    "属性视图: 显示文件属性",
    "属性视图：显示文件属性",
  ];
  const viewMatch = cmds.find(c => viewIds.includes(c.id))
    || cmds.find(c => viewNames.includes(c.name))
    || cmds.find(c => ((c.id || "").includes("properties-view") && (c.name || "").toLowerCase().includes("properties")));

  const match = listMatch || viewMatch;
  if (match) {
    await app.commands.executeCommandById(match.id);
    new Notice("已打开属性（当前笔记）", 3000);
  } else {
    new Notice("未找到“属性列表: 显示当前笔记的属性列表”或“属性视图: 显示文件属性”命令，请确认相关核心插件已启用。", 8000);
  }
} catch (e) {
  console.warn("触发属性视图命令失败", e);
  new Notice("触发属性视图命令失败，请查看控制台日志。", 6000);
}
