@DriftDatabase(tables: [
  Bills, Tasks, Memos, Contacts, PriceRecords, PetGrowth, AiLogs,

class AppDatabase extends _$AppDatabase {
  AppDatabase() : super(_openConnection());

  @override

  @override

    onCreate: (Migrator m) async {
      await m.createAll();
      // 创建FTS5虚拟表
      await customStatement('''
        CREATE VIRTUAL TABLE IF NOT EXISTS fts_notes
        USING fts5(content, md_file_path, tokenize='unicode61')
      ''');
    },
  );

  static LazyDatabase _openConnection() {
    return LazyDatabase(() async {
      final dbFolder = await getApplicationDocumentsDirectory();
      final file = File(p.join(dbFolder.path, 'symnex.db'));
      return NativeDatabase.createInBackground(file);
    });
  }
}

#### P1-03: 枢轴偏移服务 (PivotService)

**描述**: 将Obsidian中的枢轴偏移.dvjs.md逻辑迁移为Flutter原生单例服务

**核心逻辑** (从现有JS代码精确迁移):

```dart
// lib/core/services/pivot_service.dart

/// 枢轴偏移逻辑服务 - 单例
/// 核心规则: 凌晨0:00-2:59视为上一"逻辑日"
/// 对"仅日期"值(无时间部分)不应用枢轴偏移
class PivotService {
  static final PivotService _instance = PivotService._internal();
  factory PivotService() => _instance;
  PivotService._internal();

  static const int defaultPivotHour = 3;

  /// 判断是否为"仅日期"(无明确时间部分)
  bool isDateOnly(DateTime dt) {
    return dt.hour == 0 && dt.minute == 0 && dt.second == 0 && dt.millisecond == 0;
  }

  /// 将日期时间按枢轴偏移为逻辑日期
  /// 仅日期值不偏移，有时间部分且 < pivotHour 的前移一天
  DateTime logicalDate(DateTime dt, {int pivotHour = defaultPivotHour}) {
    if (isDateOnly(dt)) return DateTime(dt.year, dt.month, dt.day);
    if (dt.hour < pivotHour) {
      return DateTime(dt.year, dt.month, dt.day).subtract(const Duration(days: 1));
    }
    return DateTime(dt.year, dt.month, dt.day);
  }

  /// 获取逻辑今天的ISO日期字符串
  String logicalTodayISO({int pivotHour = defaultPivotHour}) {
    return _toISODate(logicalDate(DateTime.now(), pivotHour: pivotHour));
  }

  /// 逻辑日起始时间 (当天pivotHour:00)
  DateTime startOfLogicalDay({DateTime? base, int pivotHour = defaultPivotHour}) {
    final b = base ?? DateTime.now();
    final adjusted = b.hour < pivotHour
        ? DateTime(b.year, b.month, b.day).subtract(const Duration(days: 1))
        : DateTime(b.year, b.month, b.day);
    return adjusted.add(Duration(hours: pivotHour));
  }

  /// 逻辑日结束时间 (次日pivotHour:00)
  DateTime endOfLogicalDay({DateTime? base, int pivotHour = defaultPivotHour}) {
    return startOfLogicalDay(base: base, pivotHour: pivotHour)
        .add(const Duration(days: 1));
  }

  /// 判断某时间是否属于指定逻辑日
  bool belongsToLogicalDate(DateTime dt, String isoDate, {int pivotHour = defaultPivotHour}) {
    return _toISODate(logicalDate(dt, pivotHour: pivotHour)) == isoDate;
  }

  /// 逾期判断: 完成日 > 到期日
  bool isLateByPivot(DateTime? due, DateTime? completion, {int pivotHour = defaultPivotHour}) {
    if (due == null || completion == null) return false;
    final dueISO = _toISODate(logicalDate(due, pivotHour: pivotHour));
    final compISO = _toISODate(logicalDate(completion, pivotHour: pivotHour));
    return compISO.compareTo(dueISO) > 0;
  }

  /// 取消标识检测: "❌ YYYY-MM-DD"
  bool isCancelledLine(String text) {
    return RegExp(r'^❌\s*\d{4}-\d{2}-\d{2}').hasMatch(text);
  }

  String _toISODate(DateTime dt) {
    return '${dt.year.toString().padLeft(4, '0')}-'
           '${dt.month.toString().padLeft(2, '0')}-'
           '${dt.day.toString().padLeft(2, '0')}';
  }
}
```
