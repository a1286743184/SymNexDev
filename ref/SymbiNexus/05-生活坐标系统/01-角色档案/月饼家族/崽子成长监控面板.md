// ================= 配置区 =================
const CONFIG = {
    folder: "05-生活坐标系统/01-角色档案/月饼家族",
    motherName: "月饼", 
    sortOrder: ["老大", "老二", "老三", "老四", "老五"]
};

// 1. 引入 v2.3 库
const kitPath = '08-科技树系统/01-OB_LifeOS_Build/00-通用模块库/PuppyBoardKit.js';
const kitFile = app.vault.getAbstractFileByPath(kitPath);
if (!kitFile) { dv.paragraph(`❌ 未找到工具库: ${kitPath}`); return; }

const code = await app.vault.read(kitFile);
const scriptFunc = new Function("window", code);
scriptFunc(window);
await new Promise(r => setTimeout(r, 100)); 

const Kit = window.PuppyBoardKit;

// ================= 2. 数据处理 =================
let pages = dv.pages(`"${CONFIG.folder}"`).where(p => !p.file.name.includes("面板"));

let stats = { puppies: [], totalWeight: 0, passCount: 0 };
let chartData = { dates: new Set(), series: [] };
let renderList = [];
let motherData = null;

let maxWeight = 0;
let maxGainPct = 0;

for (let p of pages) {
    let isMother = p.file.name === CONFIG.motherName;
    let pName = p.file.name;
    let events = await Kit.logic.parseLog(app, p.file.path);
    let weights = events.filter(e => e.weight !== null).map(e => ({ weight: e.weight, date: e.date }));
    let sortedWeights = weights.sort((a, b) => b.date.localeCompare(a.date));
    
    // 图表数据 (收录所有人)
    if (sortedWeights.length > 0) {
        let weightMap = new Map();
        sortedWeights.forEach(w => {
            if (!weightMap.has(w.date)) {
                weightMap.set(w.date, w.weight);
                chartData.dates.add(w.date);
            }
        });
        chartData.series.push({ name: pName, dataMap: weightMap });
    }

    let item = {
        name: pName, path: p.file.path, isMother: isMother,
        weight: 0, gain: 0, gainPct: 0,
        isAlert: false, isChampion: false
    };

    if (sortedWeights.length > 0) {
        let curW = sortedWeights[0].weight;
        item.weight = curW;

        let gain = 0, gainPct = 0;
        if (sortedWeights.length > 1) {
            let lastW = sortedWeights[1].weight;
            gain = curW - lastW;
            if (lastW > 0) gainPct = gain / lastW;
        }
        item.gain = gain; item.gainPct = gainPct;
        
        if (isMother) {
            motherData = item;
        } else {
            // 🎯 新增过滤逻辑：仅显示当日有数据的崽子
            let lastDate = sortedWeights[0].date;
            let todayStr = moment().format("YYYY-MM-DD");
            
            // 只有当最新数据的日期是今天时，才进行统计和显示
            if (lastDate === todayStr) {
                // 崽子统计
                if (curW > maxWeight) maxWeight = curW;
                if (gainPct > maxGainPct) maxGainPct = gainPct;

                let daysOld = 0;
                if (p.birth) daysOld = Kit.logic.getDaysOld(p.birth);
                
                let status = Kit.logic.evaluateStatus(daysOld, gainPct);
                stats.totalWeight += curW;
                stats.puppies.push({ name: pName, gainPct: gainPct, gain: gain, weight: curW });
                if (status.isPass) stats.passCount++;
                
                // 预警判断
                if (gain < 0) item.isAlert = true; 
                else if (!status.isPass && sortedWeights.length > 1) item.isAlert = true; 
                
                renderList.push(item);
            }
        }
    }
}

// ================= 3. KPI 计算 =================
let kpi = { pass: "0%", avg: "--", top: "无", bot: "无", botColor: "#9ca3af" };
if (stats.puppies.length > 0) {
    kpi.avg = (stats.totalWeight / stats.puppies.length).toFixed(0);
    kpi.pass = `${((stats.passCount / stats.puppies.length) * 100).toFixed(0)}%`; 
    
    let valid = stats.puppies; 
    if (valid.length > 0) {
        // 找冠军
        valid.sort((a, b) => b.gainPct - a.gainPct);
        kpi.top = valid[0].name;
        let champItem = renderList.find(i => i.name === kpi.top);
        if (champItem) champItem.isChampion = true;
        
        // 找预警
        valid.sort((a, b) => a.weight - b.weight);
        kpi.bot = valid[0].name;   
        kpi.botColor = valid[0].gain < 0 ? "#EF4444" : "#F59E0B";
    }
}

// ================= 4. UI 渲染 =================
const container = dv.container.createEl("div");
container.className = "puppy-container"; 

// A. Header 
let samplePup = pages.find(p => p.file.name !== CONFIG.motherName);
let avgDays = samplePup ? Kit.logic.getDaysOld(samplePup.birth) : 0;
Kit.renderCockpitHeader(container, "月饼家族", avgDays);

// B. KPI 
Kit.renderCockpitConsole(container, kpi);

// C. Mother Widget
Kit.renderMotherWidget(container, motherData);

// D. List (v2.3)
Kit.renderCardList(container, renderList, maxWeight, maxGainPct, CONFIG.sortOrder);

// E. Chart 
if (chartData.series.length > 0) {
    Kit.createGrowthChart(container, chartData); 
}
